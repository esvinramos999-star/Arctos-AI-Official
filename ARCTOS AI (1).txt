from fastapi import FastAPI
from pydantic import BaseModel
import uvicorn
from inference_engine import generate_arctos_response

app = FastAPI(title="ARCTOS Backend")

class ChatRequest(BaseModel):
    message: str

class ChatResponse(BaseModel):
    response: str
    sources: list
    confidence: float

@app.post("/chat", response_model=ChatResponse)
async def chat_endpoint(payload: ChatRequest):
    result = generate_arctos_response(payload.message)
    return result

if __name__ == "__main__":
    uvicorn.run("main:app", host="0.0.0.0", port=8000, reload=True)

import { useState } from 'react';

export default function Home() {
  const [input, setInput] = useState("");
  const [response, setResponse] = useState("");

  async function sendMessage() {
    const res = await fetch("http://localhost:8000/chat", {
      method: "POST",
      headers: { "Content-Type": "application/json" },
      body: JSON.stringify({ message: input })
    });

    const data = await res.json();
    setResponse(data.response);
  }

  return (
    <div style={{
      backgroundColor: "#0b0f14",
      color: "white",
      minHeight: "100vh",
      padding: "40px",
      fontFamily: "Arial"
    }}>
      <div style={{ textAlign: "center" }}>
        <img 
          src="/arctos_logo.png" 
          alt="ARCTOS Logo" 
          style={{ width: "180px", opacity: 0.9 }}
        />
        <h1>ARCTOS</h1>
        <p>Neutral. Tactical. Fact-based.</p>
      </div>

      <div style={{ marginTop: "40px", textAlign: "center" }}>
        <input
          style={{ width: "60%", padding: "12px", fontSize: "18px" }}
          placeholder="Ask ARCTOS..."
          onChange={e => setInput(e.target.value)}
        />
        <button 
          style={{ padding: "12px 20px", marginLeft: "10px" }}
          onClick={sendMessage}>
          Send
        </button>
      </div>

      {response && (
        <div style={{
          marginTop: "40px",
          background: "#111922",
          padding: "20px",
          borderRadius: "8px"
        }}>
          <h3>Response:</h3>
          <p>{response}</p>
        </div>
      )}
    </div>
  );
}
